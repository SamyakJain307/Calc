let isSquareRoot = false;  // To track if the √ symbol is active


isSquareRoot = false;  // Reset the square root flag


// Function to append √ symbol and enable square root calculation
function appendSquareRoot() {
    if (!isSquareRoot) {  // Ensure √ is not already present
        calcInput += "√";  // Add the √ symbol to the input
        document.getElementById('calc-input').innerText = calcInput;  // Display the √ symbol
        isSquareRoot = true;  // Set the flag that square root mode is active
    }
}

function calculate() {
    try {
        // Check if √ is present in the input
        if (isSquareRoot) {
            // Extract the number after the √ symbol
            let numberAfterSquareRoot = calcInput.split("√")[1];
            
            // Calculate the square root
            calcResult = Math.sqrt(eval(numberAfterSquareRoot));
            calcResult = calcResult.toFixed(5);  // Limit the result to 5 decimal places
            
            // Display the result and reset the input
            document.getElementById('calc-input').innerText = calcResult;
            calcInput = calcResult.toString();
            isSquareRoot = false;  // Reset square root flag after calculation
        } else {
            // If no square root, calculate normally
            calcResult = eval(calcInput);  // Calculate the result using eval
            document.getElementById('calc-input').innerText = calcResult;  // Display the result
            calcInput = calcResult.toString();  // Set the input to the result for further calculations
        }
    } catch (e) {
        document.getElementById('calc-input').innerText = "Error";  // Display error if invalid calculation
    }
}
